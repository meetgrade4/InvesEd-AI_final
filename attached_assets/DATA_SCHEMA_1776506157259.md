# InvesEd AI — Data Schema

All TypeScript interfaces live in `src/types/index.ts`. All mock data lives in `src/data/`. Firestore collections mirror these schemas exactly.

## Core User & Auth

```typescript
interface User {
  id: string;                        // Firebase UID
  email: string;
  username: string;                  // Unique public handle (leaderboards)
  displayName: string;
  avatarId: string;                  // One of 20 preset illustrated avatars
  dob: string;                       // ISO date — used for age gate
  parentConsentGiven: boolean;       // Required if age < 16
  createdAt: Timestamp;
  tier: 'free' | 'pro';
  proExpiresAt?: Timestamp;

  // Gamification
  xp: number;
  level: number;                     // 1–8
  streak: number;                    // Current daily login streak (days)
  longestStreak: number;
  lastLoginDate: string;             // ISO date string
  badges: string[];                  // Array of badge IDs earned
  pinnedBadges: string[];            // Up to 3, shown on leaderboard

  // Learning
  riskProfile: RiskProfile;
  academyProgress: AcademyProgress;
  quizHistory: QuizAttempt[];
  situationRoundsCompleted: string[]; // Array of round IDs

  // Simulator
  portfolios: string[];              // Array of portfolio IDs (max 1 free, 3 pro)
  activePortfolioId: string;

  // Research
  watchlist: string[];               // Stock/fund tickers
  researchHistory: ResearchEntry[];  // Tracks all stocks/funds viewed
  researchIntelligenceScore: number; // RIS — 0-100

  // Certificates
  certificatesEarned: Certificate[];
}

type RiskProfile = {
  type: 'conservative' | 'moderate' | 'aggressive';
  label: 'Wealth Builder' | 'Balanced Investor' | 'Growth Seeker';
  score: number;                     // Raw quiz score 0-100
  completedAt: Timestamp;
  estimatedCapital: number;          // In rupees — user self-reported
  investmentGoal: 'short-term' | 'long-term' | 'both';
  knowledgeLevel: 'none' | 'basic' | 'intermediate';
}
```

## Academy & Learning

```typescript
interface AcademyProgress {
  currentModuleId: string;
  completedModules: string[];        // e.g. ['CF1', 'CF2', 'CF3']
  completedLessons: string[];        // e.g. ['CF1-L1', 'CF1-L2']
  moduleScores: Record<string, number>; // moduleId -> quiz score %
  bridgeRoundsCompleted: string[];   // Bridge round IDs between modules
  totalStudyTimeMinutes: number;
}

interface Module {
  id: string;                        // 'CF1'...'CF6', 'AE1'...'AE4'
  track: 'core' | 'advanced';
  title: string;
  description: string;
  requiredTier: 'free' | 'pro';
  lessons: Lesson[];
  finalQuizId: string;
  bridgeRoundId?: string;            // Situation Round ID triggered after completion
  simulatorFeatureUnlocked?: string; // Feature string unlocked on completion
  xpReward: number;
  prerequisites: string[];           // Module IDs that must be completed first
}

interface Lesson {
  id: string;                        // 'CF1-L1'
  moduleId: string;
  title: string;
  durationMinutes: number;
  videoUrl?: string;
  content: LessonContent[];          // Array of content blocks
  keyTakeaway: string;
  quizId: string;
  xpReward: number;
}

type LessonContent =
  | { type: 'text'; body: string }
  | { type: 'highlight'; body: string; variant: 'info' | 'warning' | 'tip' }
  | { type: 'example'; scenario: string; outcome: string }
  | { type: 'comparison'; leftLabel: string; rightLabel: string; rows: [string, string][] }
  | { type: 'calculator'; calcType: 'sip' | 'compound' | 'cagr' }

interface QuizQuestion {
  id: string;
  moduleId: string;
  text: string;
  options: string[];                 // Always 4 options
  correctIndex: number;              // 0-3
  explanation: string;               // Shown after answering
  difficulty: 'easy' | 'medium' | 'hard';
  topic: string;
}

interface QuizAttempt {
  quizId: string;
  moduleId: string;
  attemptedAt: Timestamp;
  score: number;                     // Percentage 0-100
  timeTakenSeconds: number;
  answers: number[];                 // User's selected option indices
  passed: boolean;
  xpEarned: number;
}
```

## Situation Rounds

```typescript
interface SituationRound {
  id: string;                        // e.g. 'SR-COVID-CRASH', 'SR-BORDER-ESCALATION'
  category: 'market' | 'geopolitical' | 'corruption' | 'depression' | 'sectoral';
  title: string;
  difficulty: 'beginner' | 'intermediate' | 'advanced';
  isBridgeRound: boolean;
  triggeredAfterModule?: string;     // Module ID if bridge round
  unlocksSimulatorFeature?: string;

  // Story context
  context: string;                   // 2-3 paragraph scenario setup
  marketCondition: {
    niftyChange: number;             // e.g. -35 for -35%
    description: string;
    historicalDate?: string;         // e.g. 'March 2020'
  };

  // User's virtual portfolio state at round start (injected dynamically)
  portfolioSnapshot?: PortfolioSnapshot;

  // Decision points
  choices: SituationChoice[];
  allowOpenReasoning: boolean;       // If true, user types their reasoning (AI-scored)

  // Outcomes & learning
  outcomes: Record<string, SituationOutcome>; // choiceId -> outcome
  historicalContext: string;         // What actually happened
  aiAnalysisPrompt: string;          // System prompt for Claude to evaluate reasoning
  keyLearnings: string[];

  // Scoring weights
  scoring: {
    decisionWeight: number;          // Usually 40
    reasoningWeight: number;         // Usually 40
    speedWeight: number;             // Usually 20
    timeLimitSeconds: number;        // e.g. 120
  };

  xpReward: { min: number; max: number };
}

interface SituationChoice {
  id: string;
  label: string;
  description: string;
  isOptimal: boolean;                // For scoring
  partialCredit?: number;            // 0-100 — some choices are partially right
}

interface SituationOutcome {
  immediateEffect: string;           // What happens to portfolio
  sixMonthOutcome: string;           // What historically happened
  oneYearOutcome: string;
  keyLesson: string;
  portfolioImpact: number;           // % change
}

interface SituationAttempt {
  roundId: string;
  userId: string;
  attemptedAt: Timestamp;
  choiceId: string;
  openReasoning?: string;            // User's typed reasoning
  aiScore: {
    decisionScore: number;           // 0-100
    reasoningScore: number;          // 0-100, AI-evaluated
    speedScore: number;              // 0-100
    totalScore: number;
    aiFeedback: string;              // Claude's analysis of reasoning
  };
  xpEarned: number;
}
```

## Portfolio & Simulator

```typescript
interface Portfolio {
  id: string;
  userId: string;
  name: string;                      // e.g. 'My Main Portfolio'
  createdAt: Timestamp;
  startingCapital: number;           // 100000 free, 1000000 pro
  currentCash: number;               // Remaining uninvested cash
  holdings: Holding[];
  sips: SIPSetup[];
  trades: Trade[];
  snapshots: PortfolioSnapshot[];    // Daily snapshots for chart
  behaviouralFlags: BehaviouralFlag[];
  stats: PortfolioStats;
}

interface Holding {
  ticker: string;                    // NSE ticker e.g. 'RELIANCE', or fund code
  type: 'stock' | 'mutualfund' | 'pms_virtual' | 'sif_virtual' | 'aif_virtual';
  quantity: number;                  // Shares or units
  avgBuyPrice: number;
  currentPrice: number;              // Updated daily from mock data
  investedAmount: number;
  currentValue: number;
  absoluteReturn: number;
  percentReturn: number;
  dayChange: number;
  sector?: string;
}

interface SIPSetup {
  id: string;
  fundCode: string;
  fundName: string;
  monthlyAmount: number;
  startDate: string;
  frequency: 'monthly' | 'weekly';
  status: 'active' | 'paused' | 'stopped';
  installmentsCompleted: number;
  totalInvested: number;
  currentValue: number;
  xirr: number;                      // Annualised return
  nextDueDate: string;
}

interface Trade {
  id: string;
  portfolioId: string;
  ticker: string;
  type: 'BUY' | 'SELL';
  orderType: 'market' | 'limit' | 'stoploss';
  quantity: number;
  price: number;
  totalAmount: number;
  executedAt: Timestamp;
  userAnnotation?: string;           // Optional note by user
  aiFlag?: BehaviouralFlag;         // Set if AI detected a pattern
}

interface BehaviouralFlag {
  type: 'panic_sell' | 'chasing_returns' | 'over_concentration' | 'fomo' | 'sip_lapse' | 'great_trade';
  triggeredAt: Timestamp;
  ticker?: string;
  message: string;                   // AI-generated coaching message
  historicalContext: string;         // Data backing up the flag
  acknowledged: boolean;
}

interface PortfolioSnapshot {
  date: string;                      // ISO date
  totalValue: number;
  cash: number;
  investedValue: number;
  dayReturn: number;
  totalReturn: number;
  cagr: number;
}

interface PortfolioStats {
  totalInvested: number;
  currentValue: number;
  totalReturn: number;
  cagr: number;
  bestTrade: string;
  worstTrade: string;
  panicSellCount: number;
  tradeCount: number;
  winRate: number;                   // % of profitable trades
  niftyBenchmark: number;            // Nifty CAGR over same period
  vsNifty: number;                   // Portfolio CAGR minus Nifty CAGR
}
```

interface AdvancedVehicle {
  id: string;

  whatIsThis: string;                // Plain-language explanation

  // Virtual simulation
  virtualAllocationAllowed: boolean;
  virtualManagerPersona: string;     // AI persona name for PMS

  // Performance (mock)

  threeYearReturn: number;
  benchmarkReturn: number;
}

interface VirtualAdvancedAllocation {
  id: string;
  portfolioId: string;
  vehicleId: string;
  vehicleType: 'pms' | 'sif' | 'aif';
  allocatedAmount: number;
  startDate: string;
  currentValue: number;
  feesDeducted: number;              // Simulated fees shown separately
  netReturn: number;                 // After fees

}

interface Badge {
  id: string;
  name: string;
  description: string;              // What it takes to earn it
  icon: string;                     // Emoji or SVG path name
  category: 'learning' | 'investing' | 'research' | 'consistency' | 'community' | 'pro_exclusive';
  tier: 'bronze' | 'silver' | 'gold';
  xpReward: number;
  isProExclusive: boolean;
  isSecret: boolean;                // Hidden until earned
  condition: BadgeCondition;
}

type BadgeCondition =
  | { type: 'module_complete'; moduleId: string }
  | { type: 'quiz_score'; moduleId: string; minScore: number }
  | { type: 'trade_count'; count: number }
  | { type: 'sip_setup'; count: number }
  | { type: 'hold_through_drawdown'; minDrawdown: number }
  | { type: 'research_count'; count: number }
  | { type: 'streak'; days: number }
  | { type: 'situation_round'; category: string; minScore: number }
  | { type: 'certificate_earned'; certificateId: string }
  | { type: 'leaderboard_rank'; board: string; rank: number };

interface LeaderboardEntry {
  userId: string;
  username: string;
  avatarId: string;
  level: number;
  pinnedBadges: string[];
  school?: string;
  city?: string;
  score: number;                    // Board-specific metric
  rank: number;
  change: number;                   // Rank change since last reset
  tier: 'free' | 'pro';
}

interface Certificate {
  id: string;
  type: 'foundations' | 'expert';
  platform: 'invesed' | 'coursera' | 'udemy';
  earnedAt: Timestamp;
  verificationId: string;           // Unique public verification string
  verificationUrl: string;
  courseraVerificationUrl?: string; // If Coursera co-branded
  linkedInShareUrl: string;
  studentName: string;
  modulesCompleted: string[];
  finalScore: number;
}
